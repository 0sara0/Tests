class   obj from draw

	alist = []  

	func  addobj				# func to return  list of objects by reference

			alist  +  new draw
			return  len(alist)
		
class  draw

	h   w

	func draw

		see copy("      " , 3)  	 # for  width distance
		see copy("*"  , 0 + w) + nl   	 # top  width 

		for  x  = 1 to  h   		 # height on 2 edges

			see copy("      " , 3) + "*" + copy(" " , w-2) + "*"  + nl  
		next

		see copy("      " , 3)  	 # for width distance
		see copy("*"  , 0 + w) + nl    	 # bottom  width
